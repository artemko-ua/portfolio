Проєкти

Проєкт 1: Автоматизація пентесту

Скрипт для автоматизації пентесту веб-додатків.

Код проекту 

import requests
from bs4 import BeautifulSoup
import json

def input_def(values):
    url = input("Enter the URL: ")
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    inputs = soup.find_all('input')
    
    for i, input_tag in enumerate(inputs):
        if i < len(values):
            input_tag['value'] = values[i]
    
    data = {
        "website": url,
        "inputs": {input_tag['name']: input_tag['value'] for input_tag in inputs if 'name' in input_tag.attrs}
    }
    
    with open('response.json', 'w') as f:
        json.dump(data, f, indent=4)

def main():

    package_manager = detect_package_manager()
    if package_manager == "No package manager detected":
        print("Error: No supported package manager found on your system")
        return
    
    print(f"Detected package manager: {package_manager}")
    
    setup = required_app() 
    
    for app in setup:
        try:
            print(f"Installing {app}")
            subprocess.run(f"sudo {package_manager} install -y {app}", 
                         shell=True, 
                         stdout=subprocess.PIPE, 
                         stderr=subprocess.PIPE, 
                         check=True)
        except subprocess.CalledProcessError as e:
            print(f"Error installing {app}: {e}")
            continue

if __name__ == "__main__":
    main()
